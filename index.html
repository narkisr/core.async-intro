Callbacks 

	    
; unbuffered channel acts as a rendezvous 
(def c (chan))

; nil since its empty
(take! c (fn [v] (println v)))

; trigger the print 
(put! c "hello world" (println "Data delivered!"))

Promises 

 

(def c (chan))

;!!< means: <writes into, !! blocks and not transaction safe
(future (println "done" (<!! c 42)))

; futures returns a promise
(future (println "Got!" (>!! c)))

; same as future but returns a channel
(thread (println "It works!" (<!! (thread 42))))

But!

A thread is still required

Go macro

  • A logical thread
  • Turns code to a state machine
  • Inspects channel operations
  • Parks blocking operations
  • Un-Parks when ready to run

Go 

	    
(go 42)

(<!! (go 42))

; notice the use if <!
(<!! (go (println "It works!" (<! (go 42)))))

Park vs Block

  • <!! >!! block a thread from a fixed thread pool
  • <! >! multiplexes a logical thread across blocking calls

Buffered channels

  • Back pressure mechanism
  • Producer/Consumer sync

Buffered Cont..

 
(def fbc (chan 1)) 

(go (>! fbc 1)
    (println "done"))

(go (>! fbc 2)
    (println "done")) 

(<!! fbc)
(<!! fbc)

Dropping buffers

  • Don't block
  • Discard messages

Dropping buffer

 
; sliding-buffer drops from start
(def fbc (chan (dropping-buffer 1)))

(go (>! fbc 1)
    (println "done"))

(go (>! fbc 2)
    (println "done")) 

; we will get only 1 (2 was thrown away)
(<!! fbc)

Closing 

 
(def c (chan))

(close! c)
    
; we get back nil
(<!! c)

Multiplexing 

 
(def a (chan))
(def b (chan))

(put! a 42)                    

; will return [42 channel-with-response]
(alts!! [a b])  

(alts!! [a :default :meh])

Timeout 

 
(<!! (timeout 1000))
  
; [nil timeout-channel] 
(alts!! [a (timeout 1000)]) 

; put into a -> [nil timeout-channel]
(alts!! [[a 42] (timeout 1000)])

More alts

 
(put! a 1)
(put! b 2)

; order will be random (prevent starving
(alts!! [a b]) 

; we can have priority by ordering
(alts!! [a b] :priority true)

Mult and Tap

 
(def to-mult (chan 1))
(def m (mult to-mult))

(let [c (chan 1)]
  (tap m c)
  (go (loop []
        (when-let [v (>! c)]
          (println "Got! " v)
          (recur))
        (println "Exiting!"))))

(>!! to-mult 42)
(>!! to-mult 43)

(close! to-mult)
